`_IO_2_1_stdout_+216 => ***`

def rv():
    p.recv()

    p.recvuntil(data)

    p.sendline(data)

def sd(data):
    p.send(data)

def add(idx,size):
    ru("choice: ")
    sl("1")
    rv()
    sl(str(idx))
    rv()
    sl(str(size))

    rv()
